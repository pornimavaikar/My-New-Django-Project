services:
  - type: web
    name: taskmanager
    env: python
    buildCommand: |
      cd taskmanager
      pip install -r requirements.txt
      python manage.py collectstatic --noinput
      python manage.py migrate
    startCommand: cd taskmanager && gunicorn taskmanager.wsgi:application --bind 0.0.0.0:$PORT
    envVars:
      - key: PYTHON_VERSION
        value: 3.11.11
      - key: DATABASE_URL
        fromDatabase:
          name: taskmanager_db
          property: connectionString
      - key: DJANGO_SECRET_KEY
        generateValue: true
      - key: DEBUG
        value: false

databases:
  - name: taskmanager_db
    databaseName: taskmanager
    plan: free
